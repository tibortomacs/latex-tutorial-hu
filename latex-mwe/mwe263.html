% !TEX lualatex

\documentclass{article}
\usepackage{fontspec,luacode}

\begin{luacode}
function rows(n)
  for i = 0, n do
    x = 0.1 * i 
    y = math.sin(x) 
    z = math.cos(x)
    tex.sprint(x .. " & " .. y .. " & " .. z .. " \\\\")
  end
end
\end{luacode}

\def\sincostable#1{%
  \begin{tabular}{lll}
    $x$ & $\sin(x)$ & $\cos(x)$ \\ \hline
    \directlua{rows(#1)}
  \end{tabular}}

\begin{document}

\sincostable{10}

\end{document}
